```go
package config

import (
    "fmt"
    "os"
    "path/filepath"
    "strconv"

    "github.com/BurntSushi/toml"
)

// Config holds all application configuration.
type Config struct {
    API    APIConfig    `toml:"api"`
    Output OutputConfig `toml:"output"`
    DB     DBConfig     `toml:"db"`
}

type APIConfig struct {
    Token    string `toml:"token"`
    Cooldown int    `toml:"cooldown"`
}

type OutputConfig struct {
    Dir string `toml:"dir"`
}

type DBConfig struct {
    Path string `toml:"path"`
}

// defaults sets built-in fallback values.
func defaults() Config {
    return Config{
        API:    APIConfig{Cooldown: 15},
        Output: OutputConfig{Dir: "lyrics"},
        DB:     DBConfig{Path: xdgDataPath("mxlrcsvc-go", "mxlrcsvc.db")},
    }
}

// Load reads the TOML config file at path (or XDG default if empty),
// then overlays environment variables.
func Load(path string) (Config, error) {
    cfg := defaults()
    if path == "" {
        path = xdgConfigPath("mxlrcsvc-go", "config.toml")
    }
    if _, err := os.Stat(path); err == nil {
        if _, err := toml.DecodeFile(path, &cfg); err != nil {
            return cfg, fmt.Errorf("config: decode %s: %w", path, err)
        }
    }
    applyEnvOverrides(&cfg)
    return cfg, nil
}

// applyEnvOverrides overlays MXLRC_* environment variables onto cfg.
// MXLRC_API_TOKEN, MXLRC_COOLDOWN, MXLRC_OUTPUT_DIR, MXLRC_DB_PATH
func applyEnvOverrides(cfg *Config) {
    if v := os.Getenv("MXLRC_API_TOKEN"); v != "" {
        cfg.API.Token = v
    }
    if v := os.Getenv("MXLRC_COOLDOWN"); v != "" {
        if n, err := strconv.Atoi(v); err == nil {
            cfg.API.Cooldown = n
        }
    }
    if v := os.Getenv("MXLRC_OUTPUT_DIR"); v != "" {
        cfg.Output.Dir = v
    }
    if v := os.Getenv("MXLRC_DB_PATH"); v != "" {
        cfg.DB.Path = v
    }
}

// xdgConfigPath returns ~/.config/<app>/<file> (or $XDG_CONFIG_HOME/<app>/<file>).
// Falls back to /config/<file> if /config directory exists (Docker detection).
func xdgConfigPath(app, file string) string {
    if _, err := os.Stat("/config"); err == nil {
        return filepath.Join("/config", file)
    }
    base := os.Getenv("XDG_CONFIG_HOME")
    if base == "" {
        home, _ := os.UserHomeDir()
        base = filepath.Join(home, ".config")
    }
    return filepath.Join(base, app, file)
}

// xdgDataPath returns ~/.local/share/<app>/<file> (or $XDG_DATA_HOME/<app>/<file>).
func xdgDataPath(app, file string) string {
    base := os.Getenv("XDG_DATA_HOME")
    if base == "" {
        home, _ := os.UserHomeDir()
        base = filepath.Join(home, ".local", "share")
    }
    return filepath.Join(base, app, file)
}
```

**Step 2 — Create internal/db/db.go:**

```go
package db

import (
    "context"
    "database/sql"
    "embed"
    "fmt"
    "os"
    "path/filepath"

    "github.com/pressly/goose/v3"
    _ "modernc.org/sqlite" // pure-Go SQLite driver
)

//go:embed migrations/*.sql
var migrations embed.FS

// Open opens (or creates) the SQLite database at path, applies WAL pragmas,
// and runs any pending goose migrations. Returns a ready-to-use *sql.DB.
// The caller must close the returned DB when done.
func Open(ctx context.Context, path string) (*sql.DB, error) {
    if err := os.MkdirAll(filepath.Dir(path), 0750); err != nil {
        return nil, fmt.Errorf("db: create data dir: %w", err)
    }

    db, err := sql.Open("sqlite", path)
    if err != nil {
        return nil, fmt.Errorf("db: open %s: %w", path, err)
    }

    // Configure SQLite pragmas for reliability and performance.
    pragmas := []string{
        "PRAGMA journal_mode=WAL",
        "PRAGMA foreign_keys=ON",
        "PRAGMA busy_timeout=5000",
        "PRAGMA synchronous=NORMAL",
    }
    for _, p := range pragmas {
        if _, err := db.ExecContext(ctx, p); err != nil {
            _ = db.Close()
            return nil, fmt.Errorf("db: pragma %q: %w", p, err)
        }
    }

    // Run embedded migrations via goose.
    goose.SetBaseFS(migrations)
    if err := goose.SetDialect("sqlite3"); err != nil {
        _ = db.Close()
        return nil, fmt.Errorf("db: set dialect: %w", err)
    }
    if err := goose.UpContext(ctx, db, "migrations"); err != nil {
        _ = db.Close()
        return nil, fmt.Errorf("db: migrate: %w", err)
    }

    return db, nil
}
```

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| filesystem → config | User-controlled config file path read by process |
| env vars → config | Process environment supplies token and paths |
| filesystem → DB | User-controlled DB path; file created by process |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-l30-01 | Information Disclosure | config.Load() — token stored in TOML file | accept | Config file is user-owned; chmod 0600 is user's responsibility. No process-level mitigation needed for v1. |
| T-l30-02 | Tampering | db.Open() — path injection via MXLRC_DB_PATH | mitigate | `os.MkdirAll(filepath.Dir(path), 0750)` ensures directory creation only; path stays user-controlled. No SQL path injection risk (path is DSN, not a query). Accept residual risk. |
| T-l30-03 | Denial of Service | SQLite busy_timeout=5000ms | accept | Single-user CLI tool; contention not a realistic concern. Timeout prevents indefinite hang. |
| T-l30-04 | Information Disclosure | SQL migration output via goose logs | accept | goose logs to stdout; no secrets in schema DDL. |
</threat_model>
