   ```go
   package lyrics

   import (
       "regexp"

       "golang.org/x/text/unicode/norm"

   // Package-level compiled regexes per LAYOUT-05 (D-10, was compiled per-call)
   var (
       reForbidden = regexp.MustCompile(`[\\\/:*?"<>|]`)
       reMultiDash = regexp.MustCompile(`[-]+`)
   )

   // Slugify sanitizes a string for use as a filename.
   func Slugify(s string) string {
       s = norm.NFKC.String(s)
       s = reForbidden.ReplaceAllString(s, "")
       s = reMultiDash.ReplaceAllString(s, "-")
       return strings.Trim(s, "-_")
   }
   ```
